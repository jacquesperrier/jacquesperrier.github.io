function envoyerDonnees(typeAction, nomEmploye, valeurAction, commentaire = "") { const urlScript = "TON_URL_DE_DEPLOYEMENT_GOOGLE_ICI"; const payload = { type: typeAction, // Doit être: 'heure', 'temp', 'nettoyage', 'reception' ou 'formation' employe: nomEmploye, valeur: valeurAction, note: commentaire }; fetch(urlScript, { method: 'POST', mode: 'no-cors', // Important pour Google Apps Script headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }) .then(() => { alert("✓ Enregistré dans Verdict IA"); }) .catch(err => console.error("Erreur d'envoi:", err)); }
