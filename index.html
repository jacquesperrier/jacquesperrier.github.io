const SHEET_ID = "18vFZctK9UI5NKa1XbwNm7xoYCl0B8_0uog5lXzwDpdM"; const SHEET_NAME = "Temperatures"; const EMPLOYES = { "1111": "Jean", "2222": "Marie", "3333": "Paul" }; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const pin = data.pin; const equipement = data.equipement; const temperature = parseFloat(data.temperature); if (!EMPLOYES[pin]) { return ContentService .createTextOutput(JSON.stringify({status:"error",message:"PIN invalide"})) .setMimeType(ContentService.MimeType.JSON); } const employe = EMPLOYES[pin]; const sheet = SpreadsheetApp .openById(SHEET_ID) .getSheetByName(SHEET_NAME); let statut = "OK"; if (temperature >= 4 && temperature <= 60) { statut = "ALERTE"; } sheet.appendRow([ new Date(), employe, equipement, temperature, statut ]); return ContentService .createTextOutput(JSON.stringify({status:"success"})) .setMimeType(ContentService.MimeType.JSON); } catch(err){ return ContentService .createTextOutput(JSON.stringify({status:"error",message:err})) .setMimeType(ContentService.MimeType.JSON); } }
